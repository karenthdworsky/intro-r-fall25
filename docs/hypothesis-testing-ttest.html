Data Preparation

Filter for Coastal giant salamander and remove missing length measurements.

salamander_data <- and_vertebrates %>%
  filter(species == "Coastal giant salamander") %>%
  select(section, length_1_mm) %>%
  filter(!is.na(length_1_mm), !is.na(section)) %>%
  filter(length_1_mm > 0)

Assumption Checks

A. Normality

Shapiro-Wilk test: if p > 0.05 the data are approximately normally distributed. Q-Q plots provide a visual check.

sample_sizes <- salamander_data %>% count(section)
print(sample_sizes)
# A tibble: 2 × 2
  section     n
  <chr>   <int>
1 CC       5397
2 OG       6352
tryCatch({
  normality_tests <- salamander_data %>%
    group_by(section) %>%
    shapiro_test(length_1_mm)
  print(normality_tests)
}, error = function(e) {
  for (forest_type in unique(salamander_data$section)) {
    group_data <- salamander_data %>% filter(section == forest_type)
    if (nrow(group_data) >= 3 & nrow(group_data) <= 5000) {
      test_result <- shapiro.test(group_data$length_1_mm)
      cat(sprintf("Forest type: %s, n = %d, W = %.4f, p-value = %.4f\n",
                  forest_type, nrow(group_data), test_result$statistic, test_result$p.value))
    }
  }
})

qq_plot <- salamander_data %>%
  ggplot(aes(sample = length_1_mm)) +
  geom_qq() +
  geom_qq_line() +
  facet_wrap(~section) +
  labs(title = "Q-Q Plots for Normality Check",
       subtitle = "Coastal Giant Salamander Length by Forest Type") +
  theme_minimal()

print(qq_plot)

Final Plot

clean_boxplot <- salamander_data %>%
  ggplot(aes(x = section, y = length_1_mm, fill = section)) +
  geom_boxplot(alpha = 0.8, outlier.size = 1.5, outlier.alpha = 0.6) +
  stat_summary(fun = mean, geom = "point", shape = 23, size = 4,
               fill = "white", color = "black", stroke = 1) +
  labs(
    title = "Snout-vent Length of Coastal Giant Salamanders",
    subtitle = "Comparison between forest types",
    x = "Forest Type",
    y = "Snout-vent Length (mm)"
  ) +
  theme_classic() +
  theme(
    legend.position = "none",
    plot.title = element_text(size = 14, face = "bold"),
    plot.subtitle = element_text(size = 12),
    axis.title = element_text(size = 11),
    axis.text = element_text(size = 10)
  ) +
  scale_fill_brewer(palette = "Set2")

print(clean_boxplot)
