Data Preparation

Calculate mean ice duration by water year and mean winter air temperature (Nov–Apr) by water year.

avg_icecover <- ntl_icecover %>%
  group_by(wyear = year + 1) %>%
  summarize(mean_duration = mean(ice_duration, na.rm = TRUE))

ntl_airtemp_wyear <- ntl_airtemp %>%
  mutate(wyear = if_else(month(sampledate) < 10, year, year + 1))

ntl_winter_airtemp <- ntl_airtemp_wyear %>%
  filter(lubridate::month(sampledate) %in% c(11, 12, 1:4)) %>%
  group_by(wyear) %>%
  summarize(mean_air_temp = mean(ave_air_temp_adjusted))

1. Join datasets

icecover_temp <- avg_icecover %>%
  inner_join(ntl_winter_airtemp, by = "wyear")

2. Exploratory plot

ggplot(data = icecover_temp, aes(x = mean_air_temp, y = mean_duration)) +

  labs(
